For Wheezy

$ apt-get install apt-transport-https
$ echo 'deb https://rex.linux-files.org/debian/ wheezy rex' >> /etc/apt/sources.list
$ wget -O - https://rex.linux-files.org/DPKG-GPG-KEY-REXIFY-REPO | apt-key add -
$ apt-get update
$ apt-get install rex
