  • Before and after hooks

    Copy to clipboard
       task prepare => "server1", "server2", sub {
      # do something
   };
   
   before prepare => sub {
      my ($server) = @_;
      say "Starting virtual machine...";
      run "xm start $server";
   };
   
   after prepare => sub {
      my ($server, $failed_connection) = @_;
      say "Stopping virtual machine...";
      run "xm stop $server";
   };
